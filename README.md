> github -u moondevnode -n sveltekit-blog -d "Svelekit Blog"

# Git Clone(Remote to Local)
> git clone https://github.com/moondevnode/sveltekit-blog.git

> cd sveltekit-blog

# Create Svelte Project
> pnpm create svelte .
